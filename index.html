// FastAPI health endpoint (Python)
from fastapi import FastAPI
from datetime import datetime

app = FastAPI()

@app.get("/api/health")
def health():
    checks = ["db","cache","queue"]
    return {
        "status": "ok",
        "checked": checks,
        "at": datetime.utcnow().isoformat()
    }
